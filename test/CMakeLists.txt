macro(pythonize_bool truth_var var)
  if(${truth_var})
    set(${var} True)
  else()
    set(${var} False)
  endif()
endmacro()

find_package(OpenMP QUIET)
pythonize_bool(OPENMP_FOUND CAGE_OPENMP_FOUND)

endif()

  "llvm-lit;lit;lit.py"
